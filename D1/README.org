#+title: Advent of Code 2025 - Day 1
* Puzzle [[https://adventofcode.com/2025/day/1][Day 1 - Advent of Code]]
- dial from 00 to 99 (NN), makes a *click* when reaching each number.

** Input Structure
- Sequence of *rotations* (1 per line)
  - ~L~ → Rotation to lower numbers
  - ~R~ → Rotation to higher numbers
+ Wrap it with %, as C++ → use [[https://en.wikipedia.org/wiki/Modular_arithmetic][modular arithmetic (Gauss)]] with floored division convention

- ~NN~ → number of rotations.
- You start from 50.

* Solution using C++ STD 20
- Meson is there just because I was learning it right before AoC. You can just ~g++ src/main.cpp -o aoc-2020-d1~ or whatever you feel like doing.
- I actually just used ~org-babbel~ to solve it.

#+begin_src C++ :includes <iostream> <cmath>

#define DIAL 50

template <typename T> T floor_mod(T a, T b){
    return a - b * std::floor(static_cast<double>(a) / static_cast<double>(b));
}

int right(int r){
    return floor_mod( (DIAL + r), 99 );
}

int left(int l){
    return floor_mod((DIAL - l), 99);
}

int fright(int dial, int r){
    return floor_mod( (dial + r), 99 );
}

int fleft(int dial, int l){
    return floor_mod((dial - l), 99);
}

int click(int clicker, int dial){
    int counter = 0;
    if(dial == 0){
        counter += 1;
    }
    return clicker + counter;
}

int main(){

    std::cout << "Dial: " << DIAL << std::endl;

/***/

    int r20 = right(20);
    int r60 = right(60);

    int a = floor(2);

    std::cout << "R20: " << r20 << std::endl;
    std::cout << "R60: " << r60 << std::endl;

/***/


    int l10 = left(10);
    int l90 = left(90);

    std::cout << "L10: " << l10 << std::endl;
    std::cout << "L90: " << l90 << std::endl;

/***/

    int password = 0;
    int dial_test = 50;

    dial_test = fleft(dial_test, 68);
    password = click(password, dial_test);
    dial_test = fleft(dial_test, 30);
    password = click(password, dial_test);
    dial_test = fright(dial_test, 48);
    password = click(password, dial_test);
    std::cout << dial_test << " IS ZERO\n";
    dial_test = fleft(dial_test, 5);
    password = click(password, dial_test);
    dial_test = fright(dial_test, 60);
    std::cout << dial_test << " IS ZERO\n";
    password = click(password, dial_test);
    dial_test = fleft(dial_test, 55);
    password = click(password, dial_test);
    dial_test = fleft(dial_test, 1);
    std::cout << dial_test << " IS ZERO\n";
    password = click(password, dial_test);
    dial_test = fleft(dial_test, 99);
    password = click(password, dial_test);
    dial_test = fright(dial_test, 14);
    password = click(password, dial_test);
    dial_test = fleft(dial_test, 82);
    password = click(password, dial_test);

    std::cout << "Example key: " << password << std::endl;

    return 0;
}
